Creating repo:
git init
create a repository in GitHub
git remote add origin link
git remote -v
git status
git add filename
git rm --filename  (remove the staged file)
git commit -m "file added"
git push origin master
git pull origin master  (to get the code from the origin branch)

branches:
	git branch <branchname>
	git branch (or) git branch --list 
	git checkout <branchname>  (to sitch btw branches)
	git branch -d <branchname> (to delete)
	git merge <branchname> (merge changes)
