Interface Explicit Implementation

	-> when one or more interface has similar signature matching methods and the class wants to implement all the matching methods


interface IAccounts{
	void InterfaceMethod();
}
interface IBooks {
	void InterfaceMethod();
}
class Ief : IAccounts, IBooks{
	void IAccounts.InterfaceMethod(){
		Write("Iaccount");
	}
	void IBooks.InterfaceMethod(){
		Write("IBooks");
	}
	static void Main(){
		Ief ie = new Ief();
		//to make one interface method as default,then removethe explicit implementation and make the method public.Then we can access the methodmethod with the implementing class
		
		//option 1 to call the explicit methods
		((IAccounts)ie).InterfaceMethod();
		((IBooks)ie).InterfaceMethod();

		//option 2
		IAccounts as = new Ief();
		ac.InterfaceMethod();

		IBooks bs = new Ief();
		bs.InterfaceMethod();
	}
}
