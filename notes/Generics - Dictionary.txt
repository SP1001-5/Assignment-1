Generics - Dictionary

class DictionaryEg{
	static void Main(){
		Dictioanry<int, string> dict = new Dictioanry<int, string>();
		dict.Add(1, "red");
		dict.Add(2, "blue");
		dict.Add(3, "green");
		dict.Add('a', "yellow");
 		foreach(int item in dict.Keys){
			Write(item);
		}
		foreach(string item in dict.Values){
			Write(item);
		}
		foreach(KeyValuePair<int, string> pair in dict){
			Write(pair.Key + pair.Value);
		}		
	}	
}
