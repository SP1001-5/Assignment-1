	public static void HashTableEg(){
		HashTable ht = new HashTable();
		ht.Add("E005", "Prakash");
		ht.Add("E006", "Arun");
		ht.Add("E007", "Kishore");
		ht.Add(12, "SQL");
		
		//iterate with keys
		foreach(var item in ht.Keys){
			Write(item);
		}

		//iterate with values
		foreach(var item in ht.Values){
			Write(item);
		}

		//iterate whole obj using dictionary entry
		foreach(DictionarEntry de in ht){
			Write(de.Key);
			write(de.Value);
		}

		//hashtables are used to store large record, retrieving by searching is very good with hashtable

		Write("Enter a key to search");
		string ekey = console.ReadLine();
		if(ht.ContainsKey(ekey)){
			Write(ekey+"is"+ ht[ekey]);
		}
		else{
			Write("Does not exist");
		}
	}
