Constructor Chaining

this -> it is used to invoke the other COnstructor that matches the parameter 
     -> is used for constructor chaining within the same class
     ->Pointer that points to the current instance of the same class

class Customer{
	int age;
	string custname;
	int custrating;
	public Customer : this(45){
		//Write("Customer Construction 1");
		custrating = 4;
		write(custname + custrating);
	}
	public Customer(int e):this("bob", 20){
		//Write("Construction 2");
		age = e
		write(custname + custrating);
	}
	public Customer(string s, int x){
		//Write("Customer construction 3");
		custname = s;
		custrating = x;
		write(custname + custrating);
	}
	public void showdata(){
		write(custname + custrating);
	}
}
internal class Constructionchaining{
	static void Main(){
		Customer customer = new Customer("Cust",15)
		customer.showdata();
		Customer cust = new Customer();
		customer.showdata();
		Customer cust3 = new Customer(5);
		customer.showdata();
		Console.Read();
	}
}
