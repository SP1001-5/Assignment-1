Base Derived

	->before the subclass the base class will be executed 
	->even if there is a parameterized constructor when the subclass is provoked it will invoke the empty constructor
	->if the empty constructor is not present it will show error in the subclass
	->To overcome this the parameter should be passed from subclass to base class using 'base' keyword

 	
class BaseClass{
	int bcv;
	//public BaseClass(){
		//Write("This is base class");
	//}

	public BaseClass(int a){
		bcv = a;
		Write(bcv);
	}
}

class Subclass : BaseClass{
	int scv;
	public Subclass(int b) : base(b){
		scv = b;
		write("This is sub class");
	}

	public Subclass(int x, int y):base(y){
		scv = x;
	}
}

class Base_derived{
	static void Main(){
		//BaseClass bc = new BaseClass();
		BaseClass bc2 = new BaseClass(10);
		Subclass sc = new Subclass(10);
		Console.Read();
	}

}
